# Basic server with Apollo

<Spacer margin="6rem 0 4rem 0" />

```js
const { ApolloServer, gql } = require("apollo-server")

// Graphql schema
const typeDefs = gql`
  type Query {
    hello: String
  }
`

// Resolvers
const resolvers = {
  Query: {
    hello: () => "helloworldnstuff",
  },
}

// Create server
const server = new ApolloServer({
  typeDefs,
  resolvers,
})

// Listen to server
server.listen().then(({ url }) => {
  console.log(`Server listening at ${url}`)
})
```
